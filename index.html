function startCamera() { if (navigator.mediaDevices && navigator.mediaDevices.getUserMedia) { navigator.mediaDevices.getUserMedia({ video: true }) .then(function(stream) { // Success const video = document.querySelector('video'); video.srcObject = stream; video.play(); }) .catch(function(error) { // Improve error handling if (error.name === 'NotAllowedError') { alert('Permission denied: Please allow camera access in your device settings.'); } else if (error.name === 'NotFoundError') { alert('No camera found: Please connect a camera to your device.'); } else { alert('Error accessing camera: ' + error.message); } }); } else { alert('Your browser does not support camera access.'); } }
